### Dual Push Path (Web + Capacitor)

```typescript
// lib/capacitor-push.ts
import { Capacitor } from '@capacitor/core';
import { PushNotifications } from '@capacitor/push-notifications';

export function isNativeApp(): boolean {
  return Capacitor.isNativePlatform();
}

export async function registerNativePush(merchantId: string, api: typeof import('./api').api) {
  if (!isNativeApp()) return;

  const permission = await PushNotifications.requestPermissions();
  if (permission.receive !== 'granted') return;

  await PushNotifications.register();

  PushNotifications.addListener('registration', async (token) => {
    // Send APNS token to backend — different endpoint than web push
    await api.post('/push/subscribe-native', {
      token: token.value,
      platform: 'ios',
      merchantId,
    });
  });

  PushNotifications.addListener('pushNotificationReceived', (notification) => {
    console.log('Push received:', notification);
  });

  PushNotifications.addListener('pushNotificationActionPerformed', (notification) => {
    // Handle tap — navigate to booking detail
    console.log('Push tapped:', notification);
  });
}
```

### Backend: Native push endpoint

New endpoint `POST /push/subscribe-native` stores APNS device token. Push service uses Apple Push Notification service (APNs) for native, web-push for PWA.

**Package needed:** `@parse/node-apn` or `firebase-admin` (FCM handles both iOS + web).

### Gotchas

- `npx cap sync` after every `npm run build` — copies `dist/` to native project
- iOS requires Xcode 15+ and an Apple Developer account ($99/year)
- Push notification entitlement must be enabled in Xcode signing
- `ios/` directory should be in `.gitignore` initially — add to repo after first successful Xcode build
- Capacitor uses WKWebView — test all CSS backdrop-filter on real device (Safari quirks)
- `manifest.json` stays for web/Android — Capacitor ignores it on iOS

## 5. Freemium Enforcement Points

### Backend Guards

| Endpoint | Check | FREE Limit | PRO/GRANDFATHERED |
|----------|-------|------------|-------------------|
| `POST /services` | Count merchant's active services | Max 2 | Unlimited |
| `POST /bookings` | Count merchant's bookings this calendar month | Max 50 | Unlimited |
| `GET /stripe/connect` | Plan check | ❌ Blocked | ✅ Allowed |
| `POST /push/subscribe` | Plan check | ❌ Blocked | ✅ Allowed |
| `POST /push/subscribe-native` | Plan check | ❌ Blocked | ✅ Allowed |
| `PATCH /merchants/:id` (SMS settings) | Plan check | ❌ Blocked | ✅ Allowed |
| Adapter endpoints (vehicle, fitment, etc.) | Plan check | ❌ Blocked | ✅ Allowed |
| `POST /intake-questions` | Plan check | ❌ Blocked (basic form only) | ✅ Allowed |

### Implementation: NestJS Guard

```typescript
@Injectable()
export class PlanGuard implements CanActivate {
  constructor(private readonly prisma: PrismaService) {}

  async canActivate(context: ExecutionContext): Promise<boolean> {
    const request = context.switchToHttp().getRequest();
    const merchant = request.merchant; // Set by AuthGuard
    if (!merchant) throw new UnauthorizedException();

    const requiredPlan = Reflect.getMetadata('requiredPlan', context.getHandler());
    if (!requiredPlan) return true; // No plan requirement

    if (merchant.plan === 'GRANDFATHERED' || merchant.plan === 'PRO') return true;
    if (merchant.plan === 'FREE' && requiredPlan === 'FREE') return true;

    throw new ForbiddenException('Upgrade to Pro to access this feature');
  }
}

// Usage:
@Post('subscribe')
@UseGuards(AuthGuard, PlanGuard)
@SetMetadata('requiredPlan', 'PRO')
async subscribe(@Body() body: SubscribeDto) { ... }
```

### Frontend: PlanGate component

```tsx
function PlanGate({ feature, children, fallback }: {
  feature: string;
  children: React.ReactNode;
  fallback?: React.ReactNode;
}) {
  const { merchant } = useMerchant();
  const isPro = merchant?.plan === 'PRO' || merchant?.plan === 'GRANDFATHERED';

  if (!isPro) return fallback ?? <UpgradePrompt feature={feature} />;
  return <>{children}</>;
}
```
